readNotice(){
  state.signUrl = "";
  OkApi.shared.readNotice(ReadNotice(state.noticeId , state.signUrl)).then((res) {
    if (res.code == 200) {
			// 改写isRead 状态
      state.isRead = true;
      update();
    } else {
      state.isRead = false;
      Global.showToast(res.msg);
    }
  });
}
