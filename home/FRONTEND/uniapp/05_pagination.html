然后Uniapp提供的api onReachBottom() 监控是否触底：

jsx
//触底事件
onReachBottom() { 
if (this.pageQuery.pageNum * this.pageQuery.pageSize >= this.total) {
	uni.showToast({
		title: '没有更多数据了',
		icon: 'none',
		duration: 1000
	});
	setTimeout(() => {
		uni.hideLoading()
	}, 500)
} else {
	if (this.pageQuery.pageNum <= this.pageQuery.pageNum - 1) {
		setTimeout(() => {
			uni.hideLoading()
		}, 500)
	} else {
		uni.showLoading({
			title: '加载中'
		});
		this.pageQuery.pageNum++
		this.getList()
	}
	setTimeout(() => {
		uni.hideLoading()
	}, 500)
}
