3. 使用 eventChannel.emit() ✔ #

A页面发送：

jsx
edit(item) {
		// 第二种方法注释掉
		// let obj = JSON.stringify(item)
		// uni.navigateTo({ url: '/pages/myPrepay/edit?item=' + obj })

// 来试试麻烦的eventChannel.emit()
		uni.navigateTo({
			url:`/pages/myPrepay/edit`,  
			success:(res)=>{
				res.eventChannel.emit('sendTestValue',{
						info:item
				})
			}
		})
},
